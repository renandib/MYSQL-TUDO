# BASICAMENTO O "JOIN" > SERVE PARA EXIBIR VARIAS TABELAS JUNTAS  > EX NA TABELA FILME > O ID GENERO E O ID DIRETOR > MOSTRA UM N° Q VEM DE UMA CHAVE ESTRANGEIRA > POREM PRA EU SABER MAIS DETALHES DESSE NUMERO EU POSSO USAR JOIN E MOSTRAR OS OUTROS RESULTADOS

# EX > EU QUERO SABER O GENERO DO FILME > POREM NA TABELA FILME SÓ MOSTRA O N° DO GENERO E O N° DO DIRETOR > Q VEM DE UMA CHAVE ESTRANGEIRA > ENTAO VOU USAR JOIN PARA JUNTAR AS OUTRAS TABELAS E EXIBIR O QUE EU QUERO 

# SELECIONE O NOME DOS FILMES E O SEUS GENEROS JEITO TOSCO Q NAO VAI APARECER O NOME DOS GENEROS 
SELECT nome_filme, FK_id_genero from tbl_filme; # > SOMENTE COM ISSO ELE IA ME MOSTRAR O N° DO GENERO QUE VEM DE UMA CHAVE ESTRANGEIRA > POREM EU QUERO O NOME DO GENERO AO INVES DO N° 

# OBS > SEMPRE QUANDO FOR FAZER JOIN > É BOM TER UM DIAGRAMA DAS TABELAS > PRA TER UMA NOCAO GERAL..
# OBS 2> SEMPRE É BOM DIFERENCIAR OS NOMES DOS CAMPOS NAS TABELAS > POIS SE TIVER NOMES IGUAIS > O JOIN PODE NAO SABER DIFERENCIAR AS TABELAS.. 
# AGORA VOU FAZER O JEITO CERTO > USANDO JOIN PARA PEGAR A OUTRA TABELA E USANDO O "ON" > ESSE ON > É PRA FALAR QUAL CAMPO Q TA FAZENDO CHAVE ESTRANGEIRA ENTRE AS DUAS TABELAS > OU SEJA EU TENHO Q FALAR UM CAMPO DE CADA TABELA Q TEM RELACIONAMENTO 

#PADRAO > NOMEDATABELA.NOME DOCAMPO , NOMETABELA2.NOME CAMPO2 from TBLPRINCIPAL  JOIN 2TBL  ON CAMPO 1°TBL = CAMPO 2° TBL 

SELECT tbl_filme.nome_filme, tbl_genero.genero from tbl_filme JOIN tbl_genero ON tbl_genero.Id_genero = tbl_filme.FK_id_genero; # > JEITO LERDO DE FAZER > MAS CERTO 

# AGORA VOU FAZER O MESMO ACIMA > POREM VOU "APELIDAR" AS TABELAS > PRA N TENQ FICAR REPETINDO O NOME DELAS TODA HORA 
SELECT a.nome_filme, b.genero from tbl_filme a JOIN tbl_genero b ON b.Id_genero = a.FK_id_genero; # > OBS Q AQUI EU SUBSTITIU PELA LETRA "A" 1° TBL E "B" > 2° TBL é melhor fazer assim > do q ficar digitando toda hora o nome da tabela..

# ex 2 > MOSTRE O NOME DO FILME / DURACAO / ANO LANCAMENTO / NOME GENERO / NOME DO DIRETOR / NASC DIRETOR > OBS VAI TENQ ACESSAR ESSES DADOS DE 3 TABELAS > ENTAO USA JOIN > 3 DADOS ESTAO NA 1° TABELA 

# PEGUEI 3 TABELAS USANDO JOIN E USANDO APELIDO PRA ELAS TBM.. e fazendo a relação delas com O ON 
SELECT a.nome_filme , a.duracao , a.ano_lancamento , b.genero, c.Nome_diretor , c.Dt_nascimento FROM tbl_filme a 
JOIN tbl_genero b ON b.id_genero = a.FK_id_genero 
JOIN tbl_diretor c ON a.FK_id_diretor = c.Id_diretor ;
